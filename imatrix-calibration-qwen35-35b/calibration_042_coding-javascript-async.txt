# Converting Nested Promise Chains to Async/Await with Robust Error Handling

This guide demonstrates how to refactor deeply nested Promise chains into clean `async/await` code, emphasizing robust error handling and best practices.

---

## 🔁 Before: Nested Promise Chains

```javascript
function fetchData() {
  return fetch('/api/users')
    .then(response => {
      if (!response.ok) throw new Error(`HTTP error: ${response.status}`);
      return response.json();
    })
    .then(data => {
      return fetch(`/api/users/${data[0].id}`)
        .then(res => {
          if (!res.ok) throw new Error(`User fetch failed: ${res.status}`);
          return res.json();
        })
        .then(user => {
          return fetch(`/api/users/${user.id}/posts`)
            .then(res => {
              if (!res.ok) throw new Error(`Posts fetch failed: ${res.status}`);
              return res.json();
            })
            .then(posts => {
              return { user, posts };
            })
            .catch(err => {
              console.error('Posts error:', err);
              throw err;
            });
        })
        .catch(err => {
          console.error('User fetch error:', err);
          throw err;
        });
    })
    .catch(err => {
      console.error('First fetch error:', err);
      throw err;
    });
}
```

---

## ✅ After: Async/Await with Robust Error Handling

```javascript
async function fetchData() {
  try {
    const usersRes = await fetch('/api/users');
    if (!usersRes.ok) throw new Error(`HTTP error: ${usersRes.status}`);
    const users = await usersRes.json();

    const userRes = await fetch(`/api/users/${users[0].id}`);
    if (!userRes.ok) throw new Error(`User fetch failed: ${userRes.status}`);
    const user = await userRes.json();

    const postsRes = await fetch(`/api/users/${user.id}/posts`);
    if (!postsRes.ok) throw new Error(`Posts fetch failed: ${postsRes.status}`);
    const posts = await postsRes.json();

    return { user, posts };
  } catch (error) {
    console.error('Fetch error:', error.message);
    throw error;
  }
}
```

## 🧩 Advanced Error Handling Pattern

For production-grade robustness, encapsulate error handling in reusable utilities:

```javascript
class ApiError extends Error {
  constructor(message, statusCode, data = null) {
    super(message);
    this.name = 'ApiError';
    this.statusCode = statusCode;
    this.data = data;
  }
}

async function safeFetch(url, options = {}) {
  try {
    const response = await fetch(url, options);
    if (!response.ok) {
      throw new ApiError(
        `HTTP ${response.status}: ${response.statusText}`,
        response.status,
        await response.json().catch(() => null)
      );
    }
    return response;
  } catch (err) {
    throw new ApiError(
      `Fetch failed: ${err.message}`,
      0,
      { cause: err }
    );
  }
}

async function processUserData() {
  let user, posts;
  try {
    const usersRes = await safeFetch('/api/users');
    const users = await usersRes.json();

    const userRes = await safeFetch(`/api/users/${users[0].id}`);
    user = await userRes.json();

    const postsRes = await safeFetch(`/api/users/${user.id}/posts`);
    posts = await postsRes.json();

    return { user, posts };
  } catch (error) {
    if (error instanceof ApiError) {
      console.error(`API Error (${error.statusCode}):`, error.message);
      if (error.statusCode === 404) {
        // Handle not found specifically
      } else if (error.statusCode === 500) {
        // Handle server error
      }
    } else {
      console.error('Unexpected error:', error);
    }
    throw error;
  }
}
```

## ⚡ Parallel Execution with Promise.all

```javascript
async function fetchUserDataParallel() {
  try {
    const [usersRes, profileRes] = await Promise.all([
      safeFetch('/api/users'),
      safeFetch('/api/profile')
    ]);

    const users = await usersRes.json();
    const profile = await profileRes.json();

    const userId = users[0].id;
    const [postsRes, commentsRes] = await Promise.all([
      safeFetch(`/api/users/${userId}/posts`),
      safeFetch(`/api/users/${userId}/comments`)
    ]);

    const posts = await postsRes.json();
    const comments = await commentsRes.json();

    return { users, profile, posts, comments };
  } catch (error) {
    console.error('Parallel fetch error:', error);
    throw error;
  }
}
```

## 🔁 Retry Logic with Exponential Backoff

```javascript
async function fetchWithRetry(url, retries = 3, delay = 1000) {
  for (let i = 0; i < retries; i++) {
    try {
      const response = await fetch(url);
      if (response.ok) return response;
      throw new Error(`HTTP ${response.status}`);
    } catch (err) {
      if (i === retries - 1) throw err;
      await new Promise(res => setTimeout(res, delay * Math.pow(2, i)));
    }
  }
}
```

## 📦 Complete Example with UI Integration

```javascript
class UserService {
  constructor(baseURL) {
    this.baseURL = baseURL;
  }

  async fetchUserProfile(userId) {
    try {
      const [user, posts, comments] = await Promise.all([
        this.fetchUser(userId),
        this.fetchUserPosts(userId),
        this.fetchUserComments(userId)
      ]);

      return {
        success: true,
        data: { user, posts, comments },
        message: 'User profile loaded successfully'
      };
    } catch (error) {
      return {
        success: false,
        error: {
          message: error.message || 'Failed to load profile',
          code: error.statusCode || 'UNKNOWN'
        }
      };
    }
  }

  async fetchUser(userId) {
    const res = await fetchWithRetry(`${this.baseURL}/users/${userId}`);
    return res.json();
  }

  async fetchUserPosts(userId) {
    const res = await fetchWithRetry(`${this.baseURL}/users/${userId}/posts`);
    return res.json();
  }

  async fetchUserComments(userId) {
    const res = await fetchWithRetry(`${this.baseURL}/users/${userId}/comments`);
    return res.json();
  }
}

// Usage
const userService = new UserService('https://api.example.com');

async function loadUserProfile(userId) {
  const result = await userService.fetchUserProfile(userId);
  if (result.success) {
    console.log('Profile:', result.data);
  } else {
    console.error('Error:', result.error.message);
  }
}
```

## ✅ Best Practices Summary

| Practice | Description |
|----------|-------------|
| **Try/Catch Blocks** | Wrap `async` functions to handle errors cleanly |
| **Specific Error Types** | Use custom classes like `ApiError` to differentiate error types |
| **Early Returns** | Return early on success or failure to reduce nesting |
| **Parallel Requests** | Use `Promise.all` to optimize concurrent API calls |
| **Retry Logic** | Implement exponential backoff for transient failures |
| **Logging** | Log errors with context for easier debugging |
| **Validation** | Validate HTTP status and response content before proceeding |

---

By adopting `async/await` with structured error handling, you improve code readability, maintainability, and resilience in asynchronous workflows.
